#include <iostream>
#include <vector>

using namespace std;

vector<int> v_fibo;

int recur_fibo(int i)
{
    if (i <= 1)
        return (i);
    if (v_fibo[i] == 0)
        v_fibo[i] = (recur_fibo(i - 1) + recur_fibo(i - 2)) % 1000000;
    return v_fibo[i];
}

int main(void)
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    long long n;
    cin >> n;
    v_fibo.resize(n + 1);
    fill_n(v_fibo.begin(), n + 1, 0);
    cout << recur_fibo(n);
    return (0);
}
