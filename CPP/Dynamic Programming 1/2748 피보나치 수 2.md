# 2748 피보나치 수 2
---
created at : 2023-08-27 01:20

#### 문제 풀이

```cpp
#include <iostream>

using namespace std;

long long arr[100] = {0, 1, };

long long recur_fibo(int i)
{
    if (i == 0 || i == 1)
        return (arr[i]);
    if (arr[i] == 0)
        arr[i] = recur_fibo(i - 1) + recur_fibo( i - 2);
    return arr[i];
}

int main(void)
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    int n;
    cin >> n;
    cout << recur_fibo(n);
    return (0);
}
```
